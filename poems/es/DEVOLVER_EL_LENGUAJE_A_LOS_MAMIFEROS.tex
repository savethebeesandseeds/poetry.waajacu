\settowidth{\versewidth}{DULCE DICHA OCULTA DE LOS QUE SE REBELAN SÓLOS } % Insert one of the average-sized verses, used for centering the poem

\begin{verse}[\versewidth]
{\scriptsize
%------------------------------------------------
El fondo común Waajacu recibe fondos para encontrar y producir, \\
la configuración aislada del set de genes que causa de vuelta \\
el habla en un animal, más especifícamente en los mámiferos. \\!
%------------------------------------------------
Waajacu, al riezgo de financiar conseguirlo promete entregar \\
dignamente cantidades de animales hablantes a sus donantes. \\
Y a financiar pandillas de barrio para que liberen aves.\\!
%------------------------------------------------
APOLOGÍA PRE-FRACASO, si vez la palabra fracaso, fracasas. \\
Empiricamente aquí parece que decir tres veces la palabra, \\
sana el espacio al fallo, Waajacu promete intentar bien la causa. \\!
%------------------------------------------------

%------------------------------------------------
}
\end{verse}
